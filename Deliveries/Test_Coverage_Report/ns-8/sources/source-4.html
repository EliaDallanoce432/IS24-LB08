 package it.polimi.ingsw.client.view.observers;
 
 import it.polimi.ingsw.client.model.GameFieldModel;
 import it.polimi.ingsw.client.view.StageManager;
 
 /**
  * This class implements the ModelObserver interface
  * and is responsible for observing changes in the GameFieldModel. When a change
  * occurs, it updates the GUI through the StageManager.
  * This class is instantiated and registered as an observer to the GameFieldModel
  * singleton instance upon creation.
  */
 public class GameFieldObserver implements ModelObserver {
 
     /**
      * Constructs a new GameFieldObserver and registers it as an observer
      * to the GameFieldModel instance.
      */
     public GameFieldObserver() {
         GameFieldModel.getInstance().addObserver(this);
     }
 
 
     /**
      * This method is called when the observed object is changed. It triggers
      * the update of the GUI through the StageManager.
      */
     @Override
     public void update() {
         StageManager.getCurrentViewController().updateGameBoard();
     }
 }
